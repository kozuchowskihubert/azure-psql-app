### JavaScript Mode Switching

```javascript
function switchMode(mode) {
    currentMode = mode;
    const container = document.querySelector('.container');
    
    // Remove all mode classes
    container.classList.remove('mode-basic', 'mode-pro', 'mode-advanced');
    
    // Add new mode class
    container.classList.add(`mode-${mode}`);
    
    // Update UI indicators
    updateModeTabs(mode);
    updateModeBadge(mode);
    
    // Save preference
    localStorage.setItem('trap-studio-mode', mode);
}
```

### Pattern Loading System

```javascript
function loadKickPattern(patternName) {
    const pattern = kickPatterns[patternName];
    if (!pattern) return;
    
    // Clear current pattern
    beatPattern['Kick'] = [];
    clearKickCells();
    
    // Load new pattern
    pattern.forEach(step => {
        beatPattern['Kick'].push(step);
        activateCell('Kick', step);
    });
    
    // Show success feedback
    showToast(`Loaded ${patternName} kick pattern! ğŸ¥`, 'success');
}
```
